基本用法: pf-table是一个element-ui组件封装,融合了分页,查询,表格,只需要写一些配置,就能很方便的创建查询表格,从而减少重复逻辑.

# 基础表格:

<template>
  <pf-table
      :data="tableData"
      :columns="columns"
      :auto-loading="false"
      :pagination="false"
  >
  </pf-table>
</template>

<script>
// import pfTable from '../components/pf-table/pf-table'
import PfTable from '../components/pf-table/pf-table'

export default {
  name: 'page-table',
  data () {
    return {
      tableData: [
        { date: '2016-05-02', name: '王小虎', address: '上海市普陀区金沙江路 1518 弄' },
        { date: '2016-05-04', name: '王小虎', address: '上海市普陀区金沙江路 1517 弄' },
        { date: '2016-05-01', name: '王小虎', address: '上海市普陀区金沙江路 1519 弄' },
        { date: '2016-05-03', name: '王小虎', address: '上海市普陀区金沙江路 1516 弄' }
      ]
    }
  },
  computed: {
    columns () {
      return [
        { prop: 'date', label: '日期', attrs: { width: 180 } },
        { prop: 'name', label: '姓名', attrs: { width: 180 } },
        { prop: 'address', label: '地址', attrs: { width: 180 } }
      ]
    }
  },
  components: {
    PfTable
  }
}
</script>
